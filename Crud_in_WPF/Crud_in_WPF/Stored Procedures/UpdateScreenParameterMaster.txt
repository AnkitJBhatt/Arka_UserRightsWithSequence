ALTER PROCEDURE [dbo].[UpdateScreenParameterMaster]
@prm_Id int,
@Screen_Id int,
@Name nvarchar(100),
@sequenceNumber int,
@IsActive BIT,
@IsDeleted BIT
AS
BEGIN
    UPDATE [ScreenParameterMaster] SET ScreenMasterId = @Screen_Id, Name = @Name ,IsActive = @IsActive, IsDeleted = @IsDeleted ,
    SequenceNumber = @sequenceNumber WHERE Id = @prm_Id;
End
