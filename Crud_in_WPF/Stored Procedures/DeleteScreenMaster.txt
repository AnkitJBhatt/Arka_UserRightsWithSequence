ALTER PROCEDURE [dbo].[DeleteScreenMaster]
    @id INT
AS
BEGIN
	UPDATE [ScreenMaster] SET IsDeleted = 1,
	IsActive = 0 Where Id = @id; 

	UPDATE GroupScreenMapping
	SET IsDeleted = 1, IsActive = 0, IsSelected =0
	WHERE ScreenMasterId = @id;
END;
