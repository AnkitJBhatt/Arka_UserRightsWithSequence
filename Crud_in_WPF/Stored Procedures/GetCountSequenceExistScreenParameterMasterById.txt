ALTER Procedure [dbo].[GetCountSequenceExistScreenParameterMasterById]
@id int,
@screenId int,
@sequenceNumber int
As
Begin 
    Select Count(*) From [ScreenParameterMaster] Where SequenceNumber = @sequenceNumber and Id != @id and ScreenMasterId = @screenId;
End
